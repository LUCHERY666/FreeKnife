-- Notification Animation with 2-minute duration
local function showNotification(text)
    notifText.Text = text
    notification.Visible = true
    
    notification.Position = UDim2.new(1, -350, 1, -85)
    TweenService:Create(
        notification,
        TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
        {Position = UDim2.new(1, -350, 1, -105), BackgroundTransparency = 0.2}
    ):Play()
    
    wait(120) -- 2 minutes as requested
    
    TweenService:Create(
        notification,
        TweenInfo.new(0.7, Enum.EasingStyle.Quad, Enum.EasingDirection.In),
        {Position = UDim2.new(1, -350, 1, -85), BackgroundTransparency = 0.9}
    ):Play()
    
    wait(0.7)
    notification.Visible = false
end

-- Dragging Logic
local UserInputService = game:GetService("UserInputService")
local dragging, dragInput, dragStart, startPos

frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
